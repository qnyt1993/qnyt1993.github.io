#### 3. 程序代码

    #include<reg52.h>
    #define uchar unsigned char
    #define uint  unsigned int
    
    sbit KEY1 = P3^2;
    sbit KEY2 = P3^3;
    sbit KEY3 = P3^4;
    
    sbit LED1 = P0^0;
    sbit LED2 = P0^1;
    sbit LED3 = P0^2;
    
    void main(){
        while(1){ 
            LED1 = KEY1;  //按下时，对应的led点亮
            LED2 = KEY2;
            LED3 = KEY3;
        }
    }
